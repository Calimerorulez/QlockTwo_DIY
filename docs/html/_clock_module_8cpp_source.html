150 time_t ClockModule::getNtpTime() {
151  IPAddress ntpServerIP; // NTP server's ip address
152 
153  while (ntpUDP.parsePacket() > 0); // discard any previously received packets
154 
155  Serial.println("Transmit NTP Request");
156  // get a random server from the pool
157  WiFi.hostByName(ntpServerName.c_str(), ntpServerIP);
158  Serial.print(ntpServerName);
159  Serial.print(": ");
160  Serial.println(ntpServerIP);
161  this->sendNTPpacket(ntpServerIP);
162  uint32_t beginWait = millis();
163  while (millis() - beginWait < 1500) {
164  int size = ntpUDP.parsePacket();
165  if (size >= NTP_PACKET_SIZE) {
166  Serial.println("Receive NTP Response");
167  ntpUDP.read(packetBuffer, NTP_PACKET_SIZE); // read packet into the buffer
168  unsigned long secsSince1900;
169  // convert four bytes starting at location 40 to a long integer
170  secsSince1900 = (unsigned long) packetBuffer[40] << 24;
171  secsSince1900 |= (unsigned long) packetBuffer[41] << 16;
172  secsSince1900 |= (unsigned long) packetBuffer[42] << 8;
173  secsSince1900 |= (unsigned long) packetBuffer[43];
174  return secsSince1900 - 2208988800UL;
175  }
176  }
177  Serial.println("No NTP Response :-(");
178  return 0; // return 0 if unable to get the time
179 }

185 void ClockModule::sendNTPpacket(IPAddress &address) {
186  // set all bytes in the buffer to 0
187  memset(packetBuffer, 0, NTP_PACKET_SIZE);
188  // Initialize values needed to form NTP request
189  // (see URL above for details on the packets)
190  packetBuffer[0] = 0b11100011; // LI, Version, Mode
191  packetBuffer[1] = 0; // Stratum, or type of clock
192  packetBuffer[2] = 6; // Polling Interval
193  packetBuffer[3] = 0xEC; // Peer Clock Precision
194  // 8 bytes of zero for Root Delay & Root Dispersion
195  packetBuffer[12] = 49;
196  packetBuffer[13] = 0x4E;
197  packetBuffer[14] = 49;
198  packetBuffer[15] = 52;
199  // all NTP fields have been given values, now
200  // you can send a packet requesting a timestamp:
201  ntpUDP.beginPacket(address, 123); //NTP requests are to port 123
202  ntpUDP.write(packetBuffer, NTP_PACKET_SIZE);
203  ntpUDP.endPacket();
204 }
